CREATE SEQUENCE NEWS_SEQUENCE
START WITH 1
INCREMENT BY 1
NOMAXVALUE;

CREATE TRIGGER NEWS_TRIGGER
BEFORE INSERT ON NEWS
FOR EACH ROW
BEGIN
SELECT NEWS_SEQUENCE.NEXTVAL INTO :NEW.NEWS_ID FROM DUAL;
END;
